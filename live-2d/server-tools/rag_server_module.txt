// ragMemoryServer.js - LLM记忆系统模块
const axios = require('axios');
const fs = require('fs');
const path = require('path');

// 读取配置文件（支持 YAML 格式）
let config = {};
try {
    const yamlPath = path.join(__dirname, '../config.yaml');
    const jsonPath = path.join(__dirname, '../config.json');
    
    if (fs.existsSync(yamlPath)) {
        const yamlContent = fs.readFileSync(yamlPath, 'utf8');
        config = parseYAML(yamlContent);
    } else if (fs.existsSync(jsonPath)) {
        config = JSON.parse(fs.readFileSync(jsonPath, 'utf8'));
    }
} catch (error) {
    console.warn('无法读取配置文件，使用默认配置');
}

// 简化的 YAML 解析函数
function parseYAML(yamlStr) {
    const result = {};
    const lines = yamlStr.split('\n');
    const stack = [{ obj: result, indent: -1 }];
    
    for (const line of lines) {
        const stripped = line.trimStart();
        if (!stripped || stripped.startsWith('#')) continue;
        
        const indent = line.length - stripped.length;
        while (stack.length > 1 && indent <= stack[stack.length - 1].indent) {
            stack.pop();
        }
        
        if (stripped.includes(':')) {
            const colonIndex = stripped.indexOf(':');
            const key = stripped.substring(0, colonIndex).trim();
            const value = stripped.substring(colonIndex + 1).trim();
            
            const current = stack[stack.length - 1].obj;
            
            if (value === '' || value.startsWith('#')) {
                current[key] = {};
                stack.push({ obj: current[key], indent: indent });
            } else {
                if (value === 'true') current[key] = true;
                else if (value === 'false') current[key] = false;
                else if (!isNaN(Number(value)) && value !== '') current[key] = Number(value);
                else if (value.startsWith('"') && value.endsWith('"')) current[key] = value.slice(1, -1);
                else current[key] = value;
            }
        }
    }
    
    return result;
}

// 定义记忆检索工具
const MEMORY_TOOL = {
    name: "search_memory",
    description: "从LLM的记忆系统中搜索相关的历史对话和信息",
    parameters: {
        type: "object",
        properties: {
            question: {
                type: "string",
                description: "要搜索的问题或关键词"
            },
            top_k: {
                type: "integer",
                description: "返回最相关的记忆数量，默认1",
                default: 1
            }
        },
        required: ["question"]
    }
};

// 搜索记忆
async function searchMemory(parameters) {
    const { question, top_k = 1 } = parameters;

    // 根据配置选择 URL
    const gatewayConfig = config.api_gateway || {};
    const ragConfig = config.rag || {};
    let url, headers = {};

    if (gatewayConfig.use_gateway) {
        // 使用网关模式
        url = `${gatewayConfig.base_url}/rag/ask`;
        if (gatewayConfig.api_key) {
            headers['X-API-Key'] = gatewayConfig.api_key;
        }
    } else {
        // 使用本地模式
        url = ragConfig.url || "http://127.0.0.1:8002/ask";
    }
    
    try {
        const response = await axios.post(url, {
            question: question,
            top_k: top_k
        }, { headers });
        
        const jsContent = response.data;
        let memoryContent = "";
        
        for (const passage of jsContent.relevant_passages) {
            memoryContent += passage.content + "\n";
        }
        
        return memoryContent.trim();
        
    } catch (error) {
        throw new Error(`记忆搜索失败: ${error.message}`);
    }
}

// 导出
module.exports = {
    getToolDefinitions: () => [MEMORY_TOOL],
    executeFunction: async (name, parameters) => {
        if (name !== "search_memory") {
            throw new Error(`不支持: ${name}`);
        }
        return await searchMemory(parameters);
    }
};
